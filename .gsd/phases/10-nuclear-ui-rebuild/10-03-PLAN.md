  <action>
    Generic card wrapper with glow variants (matching reference card types):

    ```tsx
    interface CardProps {
      children: React.ReactNode;
      className?: string;
      hover?: boolean;
      glow?: 'green' | 'gold' | 'red' | 'cyan' | 'none';
    }

    export function Card({ children, className = '', hover = false, glow = 'none' }: CardProps) {
      const glowMap = {
        green: 'shadow-glow-green border-neon-green/20',
        gold: 'shadow-glow-gold-top border-t-neon-gold/30',
        red: 'shadow-glow-red border-neon-red/20',
        cyan: 'shadow-glow-cyan border-neon-cyan/20',
        none: '',
      };
      return (
        <div className={`
          bg-surface backdrop-blur-xl border border-white/5 rounded-2xl
          ${hover ? 'hover:border-white/10 hover:shadow-card-hover transition-all' : ''}
          ${glowMap[glow]}
          ${className}
        `}>
          {children}
        </div>
      );
    }
    ```

    The `glow='gold'` variant is specifically for the "System Status" card from the reference, which has a warm amber top-edge glow.
