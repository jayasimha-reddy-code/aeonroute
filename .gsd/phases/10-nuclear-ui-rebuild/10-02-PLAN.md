<task id="4">
  <name>Clean up GlassCard and GlassButton primitives</name>
  <files>frontend/src/components/ui/GlassCard.tsx, frontend/src/components/ui/GlassButton.tsx</files>
  <action>
    Rewrite to use Tailwind tokens directly. Match reference image card style: semi-opaque dark panels (NOT heavily transparent).

    **GlassCard.tsx — reference-matched card surface:**
    ```tsx
    interface GlassCardProps {
      children: React.ReactNode;
      className?: string;
      glow?: 'none' | 'gold' | 'green';
      padding?: string;
      variant?: string;
    }

    export function GlassCard({ children, className = '', glow = 'none', padding, variant, ...props }: GlassCardProps) {
      const glowStyles = {
        none: '',
        gold: 'shadow-glow-gold-top border-t border-t-neon-gold/30',
        green: 'shadow-glow-green border-neon-green/20',
      };
      return (
        <div
          className={`bg-surface backdrop-blur-xl border border-white/5 rounded-2xl shadow-card ${glowStyles[glow]} ${className}`}
          {...props}
        >
          {children}
        </div>
      );
    }
    ```

    Note the `glow='gold'` variant — this is for the "System Status" card in the reference which has a distinct golden top-edge glow.

    **GlassButton.tsx — pill-shaped buttons:**
    ```tsx
    export function GlassButton({ children, className = '', variant = 'default', ...props }) {
      const variants = {
        default: 'bg-surface hover:bg-surface-hover text-white border border-white/10',
        primary: 'bg-neon-green hover:bg-neon-green/90 text-background font-semibold shadow-glow-green',
        danger: 'bg-neon-red-dim hover:bg-neon-red/20 text-neon-red border border-neon-red/20',
      };
      return (
        <button
          className={`px-4 py-2 rounded-full backdrop-blur-xl transition-all ${variants[variant]} ${className}`}
          {...props}
        >
          {children}
        </button>
      );
    }
    ```
  </action>
