    ```tsx
    import { memo } from 'react';

    import type { LucideIcon } from 'lucide-react';

    interface StatCardProps {
      title: string;
      value: number | string;
      suffix?: string;
      subtitle?: string;
      trend?: { value: string; direction: 'up' | 'down' };
      icon?: LucideIcon;
      accent?: 'emerald' | 'amber' | 'cyan' | 'red' | 'blue';
      children?: React.ReactNode; // For sparkline or progress ring
      className?: string;
    }

    const accentColors = {
      emerald: 'bg-emerald-dim text-emerald',
      amber:   'bg-amber-dim text-amber',
      cyan:    'bg-cyan-dim text-cyan',
      red:     'bg-red-dim text-red',
      blue:    'bg-blue-dim text-blue',
    };

    export default memo(function StatCard({
      title, value, suffix, subtitle, trend, icon: Icon, accent = 'emerald', children, className,
    }: StatCardProps) {
      return (
        <div className={cn(
          'bg-glass backdrop-blur-2xl border border-white/[0.05] rounded-2xl p-5',
          'transition-all duration-500 ease-out hover:-translate-y-0.5 hover:border-white/[0.10] hover:shadow-glass-hover',
          className,
        )}>
          <div className="flex items-start justify-between mb-3">
            <div className="flex items-center gap-3">
              {Icon && (
                <div className={cn('w-10 h-10 rounded-xl flex items-center justify-center', accentColors[accent])}>
                  <Icon className="w-5 h-5" />
                </div>
              )}
              <div>
                <p className="text-xs text-slate-400 uppercase tracking-wider">{title}</p>
                <div className="flex items-baseline gap-1.5 mt-1">
                  <span className="text-2xl font-bold text-white tabular-nums">
                    {typeof value === 'number' ? <AnimatedNumber value={value} /> : value}
                  </span>
                  {suffix && <span className="text-sm text-slate-400">{suffix}</span>}
                </div>
              </div>
            </div>
            {trend && (
              <span className={cn(
                'inline-flex items-center gap-1 rounded-full px-2 py-0.5 text-xs font-medium',
                trend.direction === 'up' ? 'bg-emerald-dim text-emerald' : 'bg-red-dim text-red',
              )}>
                {trend.direction === 'up' ? <TrendingUp className="w-3 h-3" /> : <TrendingDown className="w-3 h-3" />}
                {trend.value}
              </span>
            )}
          </div>
          {subtitle && <p className="text-xs text-slate-400">{subtitle}</p>}
          {children && <div className="mt-3">{children}</div>}
        </div>
      );
    });
    ```

    - Same glass card styling as all other cards
